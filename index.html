отправляем данные из локального

  1. git clone # клонируем репозитоий на свой компьютер
  2. git add . # делаем все новые файлы в этой папке отслеживаемыми и готовыми к коммиту (если нужно добавить только один файл, то вместо точки пишем его название)
  3. git commit -m "название_коммита" # создаем коммит с описанием изменений
  4. git push # отправляем данные из локального репозитория в удаленный (на github) # снова вносим какие-то изменения # возвращаемся к шагу с git add . и проходим цикл заново
  5. git pull # получить изменения с сервера и смержить их со своим кодом
